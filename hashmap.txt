    {

        HashMap<Integer, String> hashmap2
            = new HashMap<Integer, String>();
        hashmap1.put(1, "one");
        hashmap1.put(2, "two");
        hashmap1.put(3, "three");

        hashmap2.put(4, "four");
        hashmap2.put(5, "five");
        hashmap2.put(6, "six");

                           + hashmap2);
    }
}
